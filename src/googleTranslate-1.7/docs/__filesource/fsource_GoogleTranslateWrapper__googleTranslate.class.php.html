  312.     public function translate($text = '', $to, $from = '')

  331.             }
  332.             else

  335.                 $this->_toLang = $to;
  336.                 $this->_fromLang = $from;

  345.             $json = json_decode($contents, true);

  350.                 if(isset($json['responseData']['detectedSourceLanguage']))

  359.                     return $this->_translatedText;
  360.                 }
  361.                 else {

  363.                 }
  364.                 
  365.             }
  366.             else
  367.             { 
  368.                 $this->_serviceError =     $json['responseDetails'];
  369.                 return false;
  370.             }
  371.         }
  372.         else
  373.         {
  374.             return false;
  375.         }
  376.     }
